private:

     char *ptr;

    string(const string &s) : size(s.size)
    {ptr = new char[size + 1];
        strcpy(ptr, s.ptr);
    }

    ~string()
    {delete [] ptr;
    }

       friend ostream &operator <<(ostream &, const string &);
       string &operator=(const char *);
 };
